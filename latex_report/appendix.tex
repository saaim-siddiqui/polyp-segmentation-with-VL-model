\section{Experiment Commands}
\label{sec:app:commands}

\subsection{Training Commands}

\begin{lstlisting}[language=bash, caption={Training commands for all experimental configurations.}, label={lst:train_commands}]
# Vision-only baseline
python -m src.train --data_root ./data/sun \
    --experiment vision_only --epochs 100

# Full VL model (all text attributes)
python -m src.train --data_root ./data/sun \
    --experiment full --epochs 100

# Shape-only ablation
python -m src.train --data_root ./data/sun \
    --experiment text_shape_only --epochs 100

# Size-only ablation
python -m src.train --data_root ./data/sun \
    --experiment text_size_only --epochs 100

# Location-only ablation
python -m src.train --data_root ./data/sun \
    --experiment text_location_only --epochs 100

# Pathology-only ablation
python -m src.train --data_root ./data/sun \
    --experiment text_boundary_only --epochs 100
\end{lstlisting}

\subsection{Evaluation Command}

\begin{lstlisting}[language=bash, caption={Evaluation command for paired model comparison.}, label={lst:eval_command}]
python -m src.evaluate \
    --checkpoint ./checkpoints/full_vl/best.pt \
    --vision_checkpoint ./checkpoints/vision_only/best.pt \
    --data_root ./data/sun \
    --output_dir ./results/sun_evaluation \
    --mc_samples 10
\end{lstlisting}
